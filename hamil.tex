\subsection{2D Topological D-Wave Superconductor}
\label{Hamil}
2D Topological Superconductor
\begin{dmath}
H = \sum_{m,n} \left[-t\left((c_{m+1,n}^{\dag}c_{m,n}+h.c.)

$$
c_{j,l} = \sum_{<<\mathbf{k}>>} e^{(i \mathbf{k} r_{i,l})} c_{\mathbf{k}}

\begin{dmath}

 c_{k_1}c_{k_2}

 \sum_{<<k_1,k_2>>}\delta_{k_1,k_2} 
 \left[-t \left((e^{i k_1 a_m}+e^{i k_2 a_n})
		c_{k_1}^{\dag}c_{k_2}+h.c.\right) -(\mu - 4t)c_{k_1}^{\dag}c_{k_2}\right]\\

 \right)

 \sum_{<<k_1,k_2>>}\delta_{k_1,k_2}

	-t

		(
			e^{i k_1 a_m}

 \sum_{<<k_1,k_2>>}\delta_{k_1,k_2}

		(

For $k,-k$ the anticummutator  $[c_{k_1},c_{k_2} ]_+ = \delta_{k_1,k_2}$ becomes  $[c_k,c_{-k}]_+=0 = c_kc_{-k}+c_{-k}c_k \Rightarrow c_kc_{-k} = \frac{1}{2}(c_k,c_{-k}+c_k,c_{-k})= \frac{1}{2}(c_k,c_{-k}-c_{-k},c_k)$   
 \begin{dmath}

 We swap the $-c_{-k}c_{k}$ and $-c_{-k}^{\dag}c_{k}^{\dag}$ in the sum over$ k$ with $-k$ and recive:
 \begin{dmath}

Let $k a_m = k_x$ and $k a_n = k_y$
\begin{dmath}
H = \sum_{<<k>>}\left[-t 2(Cos(k_x)+Cos(k_y))-(\mu - 4t)\right] c_{k}^{\dag}c_{k}

For symetry we move $\frac{1}{2}$ of the $\left[-t 2(Cos(k_x)+Cos(k_y))-(\mu - 4t)\right] c_{k}^{\dag}c_{k}$ Term from $k\rightarrow-k$ resulting in  $\frac{1}{2}\left[-t 2(Cos(k_x)+Cos(k_y))-(\mu - 4t)\right] (c_{k}^{\dag}c_{k}-c_{-k}c_{-k}^{\dag})$
\begin{dmath}
H = \sum_{<<k>>}\frac{1}{2}\left[-t 2(Cos(k_x)+Cos(k_y))-(\mu - 4t)\right] (c_{k}^{\dag}c_{k}-c_{-k}^{\dag}c_{-k})
+\frac{1}{2}\left(
2i\Delta (Sin(k_x)+iSin(k_y)) 
c_{k}^{\dag}c_{-k}^{\dag}

This Hamiltonian has the form:
\begin{dmath}
H_{BdG}=\sum_{<<k>>}
\left[
	-\Psi_k^{\dag}

Where
\begin{dmath}
T = \left(
\begin{array}{cc}
 \frac{1}{\sqrt{\frac{2 \alpha  \left(\alpha +\sqrt{\alpha ^2+\beta  \beta ^*}\right)}{\beta  \beta ^*}+2}} & \frac{1}{\sqrt{\frac{2 \alpha  \left(\alpha -\sqrt{\alpha ^2+\beta  \beta ^*}\right)}{\beta  \beta ^*}+2}} \\
 \frac{-\alpha -\sqrt{\alpha ^2+\beta  \beta ^*}}{\beta  \sqrt{\frac{2 \alpha  \left(\alpha +\sqrt{\alpha ^2+\beta  \beta ^*}\right)}{\beta  \beta ^*}+2}} & \frac{\sqrt{\alpha ^2+\beta  \beta ^*}-\alpha }{\beta  \sqrt{\frac{2 \alpha  \left(\alpha -\sqrt{\alpha ^2+\beta  \beta ^*}\right)}{\beta  \beta ^*}+2}} \\
\end{array}
\right)
\end{dmath}
and 
$$T^{\dag} = T^{-1}$$
As $T$ is unitary $\Psi_k^{\dag}T_k^{-1}=\Psi_k^{\dag}T_k^{\dag}= (T_k\Psi_k)^{\dag}=\Phi_k^{\dag}$, thus he new creators $\Phi_k$ are still Bosons as
\begin{dmath}
	[\Phi_{k_1},\Phi_{k_2}^{\dag}]_+ =
	\left[T\Psi_k{K_1},\Psi_{k_2}^{\dag}T^{-1}\right]_+ = 
	T\Psi_{k_1}\Psi_{k_2}^{\dag}T^{-1}+\Psi_{k_2}^{\dag}T^{-1}T\Psi_{k_1}
	=
	TT^{-1}\Psi_{k_1}\Psi_{k_2}^{\dag}+\Psi_{k_2}^{\dag} \mathbb{1} \Psi_{k_1}
	=[\Psi_{k_1},\Psi_{k_2}^{\dag}]_+
\end{dmath}
The resulting diagonal Hamiltonian is:
\begin{dmath}
\sum_{<<k>>}

\begin{dmath}
\hbar \omega_k = (\alpha_k-|\beta_k|+\alpha_{-k}+|\beta_{-k}|) 
=  t (Cos(k_x)+Cos(k_y))+4t-\mu-|\Delta (Sin(k_y)-iSin(k_x))|

= 2 \left(

	\right)
\end{dmath}
